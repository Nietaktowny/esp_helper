370static inline void bmp_readout2float(int32_t* tin, uint32_t *pin, float *tout, float *pout)
371{
372 if (tin && tout)
373 *tout = (float)*tin * 0.01f;
374 if (pin && pout)

385static inline int bmp_readoutFloat(bmp_handle_t bmp, float* temperature, float* pressure)
386{
387 int32_t t = 0;
